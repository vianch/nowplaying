// Here start HUGE application with my IIFE and my namespace "huge"
var huge;
(function (huge) {
 
    //UTILITIES MODULE
    huge.Utilities = function() {
        var callRestService = function(method,restUrl,async) {
            var client;
            Eif (window.XMLHttpRequest) {
                client = new XMLHttpRequest();
            }
            client.open(method,restUrl ,async);
            client.send();
            return client;
        };
 
        return {
            callRestService: callRestService
        };
    };
    //HUGE NAVIGATION CONTROLLER MODULE
    huge.HugeNavigationController = function() {
        var utilities = huge.Utilities();
 
        function controller() {
            this.loadNavigationData();
            this.resetOnBodyClickEvent();
        }
        controller.prototype.loadNavigationData = function() {
            var callService = utilities.callRestService("GET","/rest/nav",true);
            var that = this;
            callService.onreadystatechange = function() {
                if (callService.readyState==4 && callService.status==200) {
                    that.renderHtmlNavigationMenu(JSON.parse(callService.responseText));
                }
            };
        };
 
        controller.prototype.renderHtmlNavigationMenu = function(responseData) {
            var menuHtmlString = '<ul>',
                subItemsLength;
            for(var i = 0, responseLength = responseData.items.length; i < responseLength; i++ ) {
                subItemsLength = responseData.items[i].items.length;
                menuHtmlString += (subItemsLength > 0) ? '<li onclick="hugeController.navigationClickEvent(event,this);">' : '<li>';
                menuHtmlString += '<a href="'+responseData.items[i].url+'">'+responseData.items[i].label+'</a>';
                menuHtmlString += this.generateSubMenuHtml(responseData.items[i].items) + '</li>';
            }
            menuHtmlString += '</ul>';
            document.getElementsByClassName('main-nav')[0].innerHTML = menuHtmlString + '</ul>';
        };
 
        controller.prototype.generateSubMenuHtml = function(subItems) {
            var subMenuHtmlString = '',
                subItemsLength = subItems.length;
            if(subItemsLength > 0) {
                subMenuHtmlString = '<ul>';
                for(var i = 0; i < subItemsLength; i++ ) {
                    subMenuHtmlString += '<li onclick="hugeController.openUrl(\''+subItems[i].url+'\')"><a href="'+subItems[i].url+'">'+subItems[i].label+'</a></li>';
                }
                subMenuHtmlString += '</ul>';
            }
            return subMenuHtmlString;
        };
 
        controller.prototype.navigationClickEvent = function(event,element) {
            var subMenuElement = element.getElementsByTagName('ul')[0];
            event.preventDefault();
            event.stopPropagation();
            this.resetMenuActive();
            this.resetSubMenuActive();
            element.classList.add('hg-menu-active');
            subMenuElement.classList.add('show-submenu');
            document.querySelector('.hg-traslucent-mask').style.display = 'block';
            return false;
        };
 
        controller.prototype.resetMenuActive = function(){
            var elementsLi = document.getElementsByTagName('li');
            for(var i = 0, elementLength = elementsLi.length; i<elementLength; i++) {
                elementsLi[i].classList.remove('hg-menu-active');
            };
        };
 
        controller.prototype.resetSubMenuActive = function() {
            var elementsActives = document.getElementsByClassName('show-submenu');
            for(var i = 0, elementLength = elementsActives.length; i<elementLength; i++) {
                elementsActives[i].classList.remove('show-submenu');
            };
        };
 
        controller.prototype.resetOnBodyClickEvent = function() {
            var that = this;
            document.querySelector('body').onclick = function () {
                that.resetMenuActive();
                that.resetSubMenuActive();
                document.querySelector('.hg-traslucent-mask').style.display = 'none';
            };
        };
 
        controller.prototype.openUrl = function(url) {
            window.location = url;
        };
 
        controller.prototype.openSidebarEvent = function(element){
            document.querySelector(element).onclick = function () {
                document.querySelector('.hg-sidebar').style.display = 'block';
            }
        };
 
 
        return controller;
    };
 
    getHugeController =  huge.HugeNavigationController();
    hugeController = new getHugeController();
 
 
})(huge || (huge = {}));
